00074 /*
00075 [STRUCTURE] 
00076 [
00077     [DESCRIPTION]
00078     Structure to define a parameter for a DSP unit.
00079 
00080     [REMARKS]
00081     Members marked with [in] mean the variable can be written to.  The user can set the value.<br>
00082     Members marked with [out] mean the variable is modified by FMOD and is for reading purposes only.  Do not change this value.<br>
00083     <br>
00084     The step parameter tells the gui or application that the parameter has a certain granularity.<br>
00085     For example in the example of cutoff frequency with a range from 100.0 to 22050.0 you might only want the selection to be in 10hz increments.  For this you would simply use 10.0 as the step value.<br>
00086     For a boolean, you can use min = 0.0, max = 1.0, step = 1.0.  This way the only possible values are 0.0 and 1.0.<br>
00087     Some applications may detect min = 0.0, max = 1.0, step = 1.0 and replace a graphical slider bar with a checkbox instead.<br>
00088     A step value of 1.0 would simulate integer values only.<br>
00089     A step value of 0.0 would mean the full floating point range is accessable.<br>
00090 
00091     [PLATFORMS]
00092     Win32, Win64, Linux, Linux64, Macintosh, Xbox360, PlayStation 2, PlayStation Portable, PlayStation 3, Wii, Solaris, iPhone
00093 
00094     [SEE_ALSO]    
00095     System::createDSP
00096     DSP::setParameter
00097 ]
00098 */

00100 {
00101     float       min;                                /* [in] Minimum value of the parameter (ie 100.0). */
00102     float       max;                                /* [in] Maximum value of the parameter (ie 22050.0). */
00103     float       defaultval;                         /* [in] Default value of parameter. */
00104     char        name[16];                           /* [in] Name of the parameter to be displayed (ie "Cutoff frequency"). */
00105     char        label[16];                          /* [in] Short string to be put next to value to denote the unit type (ie "hz"). */
00106     const char *description;                        /* [in] Description of the parameter to be displayed as a help item / tooltip for this parameter. */
